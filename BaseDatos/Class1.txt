CREATE DATABASE luis;


CREATE TABLE public.alumnos
(
    "Fecha_nacimiento" date,
    "Nombre" character varying(25),
    "Direccion" character varying(30),
    "Boleta" integer,
    id serial,
    activo boolean,
    PRIMARY KEY (id)
);


CREATE TABLE public.docente
(
	"Nombre_1" character varying(25),
	"Nombre_2" character varying(25),
	"Apellido_Paterno" character varying(25),
	"Apellido_Materno" character varying(25),
	"#emp" serial,
	"Fecha_nacimiento" date,
	"CURP"  character varying(25) unique NOT NULL,
	CONSTRAINT Pk_Docente PRIMARY KEY ("#emp") 
);



CREATE TABLE public.Materia
(
    "Codigo" integer,
    "Nombre" character varying(25),
    "Creditos" float,
    PRIMARY KEY ("Codigo")
);


CREATE TYPE genero AS ENUM(
	 'Femenino',
	 'Masculino',
	 'femenino',
	 'masculino',
 	 'FEMENINO',
	 'MASCULINO'
);

CREATE TYPE direccion AS ENUM(
	 'Calle' varchar(100),
	 'Ciudad' varchar(100),
	 'Codigo_Postal' varchar(50),
	 'Colonia' varchar(100)
);

/*   Crea sequencia  */
create sequence boleta_seq as bigint
start with 2019000000
minvalue 2019000000
maxvalue 2999999999
increment by 1;
